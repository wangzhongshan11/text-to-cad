def rotation_matrix(rx, ry, rz):
    cx, cy, cz = np.cos([rx, ry, rz])
    sx, sy, sz = np.sin([rx, ry, rz])

    Rz = np.array([
        [cz, -sz, 0],
        [sz, cz, 0],
        [0, 0, 1]
    ])

    Ry = np.array([
        [cy, 0, sy],
        [0, 1, 0],
        [-sy, 0, cy]
    ])

    Rx = np.array([
        [1, 0, 0],
        [0, cx, -sx],
        [0, sx, cx]
    ])

    return Rz @ Ry @ Rx

def constraints(x, bodyA, bodyB):

    a_tx, a_ty, a_tz, a_rx, a_ry, a_rz, \
    b_tx, b_ty, b_tz, b_rx, b_ry, b_rz = x

    residuals = []

    # 计算旋转矩阵
    Ra = rotation_matrix(a_rx, a_ry, a_rz)
    Rb = rotation_matrix(b_rx, b_ry, b_rz)

    # --------- COAXIAL_Z ----------
    # 1. XY 对齐
    residuals.append(a_tx - b_tx)
    residuals.append(a_ty - b_ty)

    # 2. Z轴方向一致
    za = Ra @ np.array([0, 0, 1])
    zb = Rb @ np.array([0, 0, 1])
    residuals.extend(za - zb)

    # --------- CONTACT_Z ----------
    residuals.append(a_tz - (b_tz + bodyB.height))

    return np.array(residuals)
